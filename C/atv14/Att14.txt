int comparar(int num1,int num2){
  int count=0,i;
  
  if (num1 >= num2){
    while(num1>=num2) {
      num1=num1-num2;
      count++;
     } return count;    
  }else{
    for(i = 1; i <= num1; i++){
     count += num2;
   }return count;
  }
}

}
